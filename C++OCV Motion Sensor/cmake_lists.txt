cmake_minimum_required(VERSION 3.20)
project(OpenCVRecorder LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find OpenCV via its CMake config package.
# If CMake can't find it automatically, pass -DOpenCV_DIR="C:/opencv/build"
# (the folder that contains OpenCVConfig.cmake).
find_package(OpenCV REQUIRED)

add_executable(recorder Recording.cpp)

# Include + link the OpenCV libraries
target_include_directories(recorder PRIVATE ${OpenCV_INCLUDE_DIRS})
target_link_libraries(recorder PRIVATE ${OpenCV_LIBS})

# Optional: set output dir tidy
set_target_properties(recorder PROPERTIES
  RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

# Note: Ensure the runtime DLLs are discoverable at run-time on Windows.
# Easiest: add C:\\opencv\\build\\x64\\vc16\\bin to your PATH (already recommended).
